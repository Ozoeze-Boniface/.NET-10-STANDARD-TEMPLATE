PostgreSQL cleanup
SELECT pg_terminate_backend(pg_stat_activity.pid)
FROM pg_stat_activity
WHERE pg_stat_activity.datname = 'keyrails_banking_api_db'
  AND pid <> pg_backend_pid();

drop database "keyrails_banking_api_db";
